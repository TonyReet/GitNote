    final Map<String, Widget> screens = {
      'Lines': Lines(),
      'Circles': Circles(),
      'Polygon': Polygon(),
      'Spiral': Spiral(),
      'Planets': Planets(),
    };

    List<Widget> children = [];
    screens.forEach((k, v) {
      children.add(
        ListTile(
          title: Text(k),
          onTap: () => Navigator.push(
            context,
            MaterialPageRoute(builder: (context) => v),
          ),
        ),
      );
    });
